📝 Créer un compte Supabase

Supabase est une plateforme gratuite qui fournit une base de données PostgreSQL hébergée.

  1. Allez sur https://supabase.com
  2. Cliquez sur "Start your project"
  3. Option Gratuite : Connectez-vous avec votre compte GitHub
  4. Sinon, cliquez sur "Sign up with email" et créez un compte
Astuce : La connexion via GitHub est la plus rapide. Sinon, utilisez votre email.
2

🏗️ Créer un nouveau projet

Une fois connecté, vous arrivez sur le dashboard Supabase.

  1. Cliquez sur le bouton "New project" (bleu, en haut à droite)
  2. Name : Donnez un nom à votre projet, par exemple : clinique-refontiq
  3. Database Password : Choisissez un mot de passe fort et notez-le quelque part !
  4. Region : Choisissez West Europe (Ireland) ou Frankfurt (proche de l'Afrique de l'Ouest)
  5. Pricing Plan : Laissez "Free Plan" (c'est gratuit)
  6. Cliquez sur "Create new project"
Patientez environ 2 minutes - Le temps que Supabase crée votre base de données.
3

🔑 Copier vos clés de connexion

Une fois le projet créé, vous devez récupérer les clés pour connecter votre application.

  1. Dans la barre latérale gauche, cliquez sur "Project Settings" (⚙️ icône)
  2. Cliquez sur "API" dans le menu

Vous allez voir 2 informations importantes :

🗄️ Exécuter le code SQL de création des tables

Il faut maintenant créer les tables de votre base de données.

  1. Dans la barre latérale gauche, cliquez sur "SQL Editor" (📝 icône)
  2. Cliquez sur "New query"

-- =====================================
-- STRUCTURE COMPLÈTE DE LA BASE DE DONNÉES
-- Clinique Refontiq
-- =====================================

-- 1. Table des utilisateurs
CREATE TABLE users (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  email TEXT UNIQUE NOT NULL,
  nom TEXT NOT NULL,
  prenom TEXT NOT NULL,
  role TEXT NOT NULL CHECK (role IN ('medecin', 'infirmiere', 'receptionniste')),
  specialite TEXT,
  telephone TEXT,
  avatar_url TEXT,
  est_connecte BOOLEAN DEFAULT FALSE,
  derniere_activite TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Table des patients
CREATE TABLE patients (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  nom TEXT NOT NULL,
  prenom TEXT NOT NULL,
  date_naissance DATE,
  sexe TEXT CHECK (sexe IN ('M', 'F')),
  telephone TEXT,
  email TEXT,
  adresse TEXT,
  groupe_sanguin TEXT,
  allergies TEXT,
  maladies TEXT,
  medecin_traitant TEXT,
  notes TEXT,
  statut TEXT DEFAULT 'actif' CHECK (statut IN ('actif', 'inactif')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Table des consultations
CREATE TABLE consultations (
  id SERIAL PRIMARY KEY,
  patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
  medecin_id UUID REFERENCES users(id),
  type TEXT CHECK (type IN ('consultation', 'urgence', 'suivi', 'bilan', 'vaccination')),
  date DATE NOT NULL,
  heure TIME NOT NULL,
  duree INTEGER DEFAULT 30,
  priorite TEXT DEFAULT 'normal' CHECK (priorite IN ('normal', 'haute', 'urgente')),
  motif TEXT,
  symptomes TEXT,
  historique TEXT,
  notes TEXT,
  diagnostic TEXT,
  traitement TEXT,
  status TEXT DEFAULT 'reception' CHECK (status IN ('reception', 'infirmerie', 'pret', 'en_consultation', 'termine', 'annule')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4. Table des rendez-vous
CREATE TABLE rendez_vous (
  id SERIAL PRIMARY KEY,
  patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
  medecin_id UUID REFERENCES users(id),
  date DATE NOT NULL,
  heure TIME NOT NULL,
  duree INTEGER DEFAULT 30,
  type TEXT CHECK (type IN ('consultation', 'suivi', 'bilan', 'vaccination', 'urgence')),
  motif TEXT,
  notes TEXT,
  statut TEXT DEFAULT 'en_attente' CHECK (statut IN ('en_attente', 'confirme', 'annule', 'termine', 'manquǸ')),
  alerte_vue BOOLEAN DEFAULT FALSE,
  rappel TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 5. Table des paiements
CREATE TABLE paiements (
  id SERIAL PRIMARY KEY,
  patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
  montant INTEGER NOT NULL,
  type TEXT NOT NULL, -- Consultation, Acte, Pharmacie
  methode_paiement TEXT CHECK (methode_paiement IN ('Espèces', 'Mobile Money', 'Carte', 'Assurance')),
  statut TEXT DEFAULT 'Payé' CHECK (statut IN ('Payé', 'En attente', 'Annulé')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 6. Table des stocks / Inventaire
CREATE TABLE stocks (
  id SERIAL PRIMARY KEY,
  nom TEXT NOT NULL,
  categorie TEXT NOT NULL CHECK (categorie IN ('Médicament', 'Matériel', 'Consommable')),
  quantite INTEGER NOT NULL DEFAULT 0,
  seuil_alerte INTEGER NOT NULL DEFAULT 10,
  fournisseur TEXT,
  prix_unitaire DECIMAL(10,2),
  date_expiration DATE,
  emplacement TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 7. Table des mouvements de stock (Historique)
CREATE TABLE stock_mouvements (
  id SERIAL PRIMARY KEY,
  produit_id INTEGER REFERENCES stocks(id) ON DELETE CASCADE,
  type_mouvement TEXT NOT NULL CHECK (type_mouvement IN ('entree', 'sortie')),
  quantite INTEGER NOT NULL,
  motif TEXT,
  utilisateur TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 8. Table des soins / Kits de soins
CREATE TABLE soins (
  id SERIAL PRIMARY KEY,
  nom TEXT NOT NULL,
  description TEXT,
  produits_requis JSONB DEFAULT '[]'::jsonb,
  prix_unitaire INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 9. Table des constantes (DME Avancé)
CREATE TABLE constantes (
  id SERIAL PRIMARY KEY,
  patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
  infirmiere_id UUID REFERENCES users(id),
  temperature DECIMAL(4,1),
  tension_sys INTEGER,
  tension_dia INTEGER,
  poids DECIMAL(5,2),
  pouls INTEGER,
  spo2 INTEGER,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 10. Table du catalogue d'actes (Interventions)
CREATE TABLE actes_medicaux (
  id SERIAL PRIMARY KEY,
  nom TEXT NOT NULL,
  prix INTEGER NOT NULL,
  categorie TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 11. Table des messages
CREATE TABLE messages (
  id SERIAL PRIMARY KEY,
  expediteur_id UUID REFERENCES users(id),
  destinataire_id UUID REFERENCES users(id),
  sujet TEXT NOT NULL,
  contenu TEXT NOT NULL,
  lu BOOLEAN DEFAULT FALSE,
  priorite TEXT DEFAULT 'normale' CHECK (priorite IN ('normale', 'haute', 'urgente')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 12. Table des alertes
CREATE TABLE alertes (
  id SERIAL PRIMARY KEY,
  type TEXT NOT NULL CHECK (type IN ('stock', 'patient', 'rendez_vous', 'systeme', 'rapport')),
  message TEXT NOT NULL,
  priorite TEXT DEFAULT 'basse' CHECK (priorite IN ('haute', 'moyenne', 'basse')),
  lu BOOLEAN DEFAULT FALSE,
  lien TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
